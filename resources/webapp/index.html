Example queries

Create Graph under Version Control

CREATE GRAPH <test>

Select Query

SELECT * 
FROM <test> REVISION "23"
WHERE {
	?s ?p ?o. 
}

Select Query - Multiple Graphs

SELECT * 
FROM <test> REVISION "MASTER"
FROM <test2> REVISION "23"
WHERE {
	?s ?p ?o. 
}

Branching

USER "mgraube"

BRANCH GRAPH <test> REVISION "2" TO "unstable"
