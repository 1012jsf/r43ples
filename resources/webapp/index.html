Select Query - Multiple Graphs

SELECT ?s ?p ?o 
FROM <http://test.com/r43ples-dataset-1> REVISION "master"
FROM <http://test.com/r43ples-dataset-2> REVISION "2"
WHERE {
	?s ?p ?o. 
}
